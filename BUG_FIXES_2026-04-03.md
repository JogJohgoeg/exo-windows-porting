**After:**
```python
async def remove_node(self, node_id: str) -> None:
    """Remove a node from the load balancer
    
    Notes:
        Silently ignores if node doesn't exist (no-op)
    """
    
    if node_id not in self.nodes:
        print(f"⚠️ Node {node_id} does not exist, skipping removal")
        return
    
    # Clean up health tracking with safe pop
    self.node_health.pop(node_id, None)  # Safe!
    self.node_failures.pop(node_id, None)  # Safe!
    
    del self.nodes[node_id]
```

### **Bug #10: Async Callback Execution in Non-Async Context**
**Location:** `network/discovery.py` - `_on_peer_lost()` method  
**Severity:** Medium (Callback Failure)  
**Description:** The callback execution relied on `asyncio.get_event_loop().is_running()`, which could return False when called from background threads, preventing callbacks from being triggered.

**Before:**
```python
def _on_peer_lost(self, name: str) -> None:
    # ... peer removal logic
    
    if self.on_peer_lost and asyncio.get_event_loop().is_running():
        asyncio.create_task(self.on_peer_lost(node_id))  # Might not execute!
```

**After:**
```python
def _on_peer_lost(self, name: str) -> None:
    # ... peer removal logic
    
    if self.on_peer_lost:
        try:
            loop = asyncio.get_running_loop()
            # In running event loop - schedule task directly
            asyncio.create_task(self.on_peer_lost(node_id))
        except RuntimeError:
            # No running event loop - use thread to run async callback
            import threading
            threading.Thread(
                target=lambda: self._run_callback_async(node_id),
                daemon=True
            ).start()

async def _run_callback_async(self, node_id: str) -> None:
    """Run async callback in background thread"""
    try:
        await self.on_peer_lost(node_id)
    except Exception as e:
        print(f"❌ Error in peer lost callback for {node_id}: {e}")
```

**Impact:** Ensured callbacks execute reliably regardless of calling context.

### **Bug #12: GPU Detection Parsing Errors**
**Location:** `exo_windows_porting/backend/backend_utils.py` - `detect_hardware()` function  
**Severity:** High (Crash Risk)  
**Description:** The NVIDIA GPU detection would crash if `nvidia-smi` output format didn't match expected structure, particularly when `memory.total` field was missing or malformed.

**Before:**
```python
for line in result.stdout.strip().split("\n"):
    parts = [p.strip() for p in line.split(",")]
    if len(parts) >= 3:
        device_id = int(parts[0])
        name = parts[1]
        memory_mb = int(parts[2].replace(" MiB", ""))  # Could crash!
        
        info.nvidia_devices.append({...})
```

**After:**
```python
for line in result.stdout.strip().split("\n"):
    parts = [p.strip() for p in line.split(",")]
    
    if len(parts) >= 3:
        try:
            device_id = int(parts[0])
            name = parts[1]
            memory_mb_str = parts[2].replace(" MiB", "").replace(" MB", "")
            
            # Validate memory field before parsing
            if memory_mb_str and memory_mb_str.isdigit():
                memory_mb = int(memory_mb_str)
                
                info.nvidia_devices.append({...})
        except (ValueError, IndexError):
            print(f"⚠️ Skipping malformed nvidia-smi line: {line}")
            continue
    elif len(parts) == 2 and parts[0].strip():
        # Handle case where memory.total is not available
        try:
            device_id = int(parts[0])
            name = parts[1]
            
            info.nvidia_devices.append({
                "id": device_id,
                "name": name,
                "memory_total_mb": 0  # Unknown
            })
        except ValueError:
            pass
```

**Impact:** Prevented crashes from malformed GPU detection output.
